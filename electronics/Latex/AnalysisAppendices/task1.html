def Increase(x, A):
    return A**x / A

def Decrease(x, A):
    return A**-(x-1) / A

xRange = np.linspace(0, 1, 100)

A=10
increasing = [Increase(x, A) for x in xRange]
decreasing = [Decrease(x, A) for x in xRange]

plt.plot(xRange, increasing)
plt.plot(xRange, decreasing)
